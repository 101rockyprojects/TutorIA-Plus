<script>
// QD: array de arrays, uno por sección
// Cada pregunta: { q: "texto", o: ["A","B","C","D"], a: índice_correcto, e: "explicación" }
const QD=[
  // Sección 0
  [
    {q:"[Pregunta 1]?",o:["[Op A]","[Op B]","[Op C]","[Op D]"],a:0,e:"✅ [Explicación pedagógica]"},
    // ... 6-7 preguntas
  ],
  // Sección 1
  [
    {q:"[Pregunta 1]?",o:["[Op A]","[Op B]","[Op C]","[Op D]"],a:1,e:"✅ [Explicación pedagógica]"},
    // ...
  ],
  // ... más secciones
];

// ── LÓGICA FIJA — NO MODIFICAR ──
const ST=QD.map(a=>({cur:0,sc:0,sel:null,conf:false,tot:a.length}));
function showS(i){document.querySelectorAll('.sb').forEach((e,j)=>e.classList.toggle('active',j===i));document.querySelectorAll('.nav-tab').forEach((e,j)=>e.classList.toggle('active',j===i));window.scrollTo({top:0,behavior:'smooth'});}
function rQ(i){const s=ST[i];const cEl=document.getElementById('c'+i);const bar=document.getElementById('b'+i);const qEl=document.getElementById('q'+i);const rEl=document.getElementById('r'+i);if(!cEl)return;document.getElementById('t'+i).textContent=s.tot;if(s.cur>=s.tot){qEl.innerHTML='';rEl.style.display='block';const p=Math.round(s.sc/s.tot*100);let em='😅',m='¡No pasa nada! Repasa y vuelve a intentarlo.';if(p===100){em='🏆';m='¡Perfecto! ¡Eres un genio!';}else if(p>=70){em='🌟';m='¡Muy bien! Repasa las que fallaste.';}else if(p>=50){em='👍';m='Bien, pero puedes mejorar.';}rEl.innerHTML=`<span class="re">${em}</span><h4>${m}</h4><div class="sb2">${s.sc}/${s.tot} (${p}%)</div><p>¡Sigue practicando! 💪</p><button class="rb" onclick="reQ(${i})">🔄 Reintentar</button>`;bar.style.width='100%';return;}rEl.style.display='none';cEl.textContent=s.cur+1;bar.style.width=`${(s.cur/s.tot)*100}%`;const q=QD[i][s.cur];s.sel=null;s.conf=false;qEl.innerHTML=`<div class="qt">${q.q}</div><div class="ol" id="ol${i}">${q.o.map((o,j)=>`<button class="ob" id="o${i}${j}" onclick="selO(${i},${j})">${o}</button>`).join('')}</div><button class="cb" id="cb${i}" onclick="confA(${i})" disabled>Confirmar respuesta ✔</button><div class="fb" id="fb${i}"></div><button class="nb" id="nb${i}" onclick="nextQ(${i})">Siguiente pregunta →</button>`;}
function selO(i,j){const s=ST[i];if(s.conf)return;s.sel=j;document.querySelectorAll(`#ol${i} .ob`).forEach((b,k)=>b.classList.toggle('sel',k===j));document.getElementById('cb'+i).disabled=false;}
function confA(i){const s=ST[i];if(s.conf||s.sel===null)return;s.conf=true;const q=QD[i][s.cur];const ok=s.sel===q.a;if(ok)s.sc++;document.querySelectorAll(`#ol${i} .ob`).forEach((b,k)=>{b.disabled=true;if(k===q.a)b.classList.add('cor');if(k===s.sel&&!ok)b.classList.add('wro');});const fb=document.getElementById('fb'+i);fb.className=`fb ${ok?'cor':'wro'}`;fb.innerHTML=`${ok?'✅ ¡Correcto!':'❌ Incorrecto.'} ${q.e}`;fb.style.display='block';document.getElementById('cb'+i).disabled=true;document.getElementById('nb'+i).style.display='inline-block';}
function nextQ(i){ST[i].cur++;rQ(i);}
function reQ(i){ST[i].cur=0;ST[i].sc=0;document.getElementById('r'+i).style.display='none';rQ(i);}
window.addEventListener('scroll',()=>document.getElementById('stb').classList.toggle('v',window.scrollY>300));
document.addEventListener('DOMContentLoaded',()=>{for(let i=0;i<QD.length;i++)rQ(i);});
</script>
